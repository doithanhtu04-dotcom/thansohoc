document.addEventListener("DOMContentLoaded", function () { const form = document.getElementById("numerologyForm"); const nameInput = document.getElementById("name"); const birthInput = document.getElementById("birthdate"); if (!form) { console.error("❌ numerologyForm NOT FOUND"); return; } form.addEventListener("submit", function (e) { e.preventDefault(); if (!nameInput || !birthInput) { alert("Lỗi input! Kiểm tra lại ID name và birthdate trong HTML."); return; } let fullName = nameInput.value.trim(); let birthdate = birthInput.value; if (!fullName || !birthdate) { alert("Vui lòng nhập đủ thông tin!"); return; } if (!birthdate.includes("-")) { alert("Ngày sinh không hợp lệ!"); return; } // ------------------------------ // HÀM RÚT GỌN SỐ // ------------------------------ function reduceNumber(num) { while (num > 9 && num !== 11 && num !== 22) { num = num.toString().split("").reduce((a, b) => a + parseInt(b), 0); } return num; } // ------------------------------ // BẢNG CHỮ -> SỐ // ------------------------------ const letterMap = { A:1, J:1, S:1, B:2, K:2, T:2, C:3, L:3, U:3, D:4, M:4, V:4, E:5, N:5, W:5, F:6, O:6, X:6, G:7, P:7, Y:7, H:8, Q:8, Z:8, I:9, R:9 }; // ------------------------------ // TÍNH TỪ TÊN // ------------------------------ function calcFromName(name, vowelsOnly = false, consonantsOnly = false) { let sum = 0; name = name.toUpperCase().replace(/[^A-Z]/g, ""); const vowels = ["A", "E", "I", "O", "U"]; name.split("").forEach(letter => { if (!letterMap[letter]) return; if (vowelsOnly && vowels.includes(letter)) sum += letterMap[letter]; else if (consonantsOnly && !vowels.includes(letter)) sum += letterMap[letter]; else if (!vowelsOnly && !consonantsOnly) sum += letterMap[letter]; }); return reduceNumber(sum); } // ------------------------------ // TÍNH TỪ NGÀY SINH // ------------------------------ function calcFromBirth(birth) { let digits = birth.replaceAll("-", "").split(""); let sum = digits.reduce((a, b) => a + parseInt(b), 0); return reduceNumber(sum); } // ------------------------------ // TÍNH TOÁN // ------------------------------ const soul = calcFromName(fullName, true); const personality = calcFromName(fullName, false, true); const destiny = calcFromName(fullName); const thinking = calcFromBirth(birthdate); const attitude = reduceNumber( parseInt(birthdate.split("-")[2]) + parseInt(birthdate.split("-")[1]) ); const maturity = reduceNumber(destiny + thinking); // ------------------------------ // LƯU LOCAL STORAGE // ------------------------------ localStorage.setItem("soul", soul); localStorage.setItem("personality", personality); localStorage.setItem("destiny", destiny); localStorage.setItem("attitude", attitude); localStorage.setItem("thinking", thinking); localStorage.setItem("maturity", maturity); window.location.href = "result.html"; }); });
